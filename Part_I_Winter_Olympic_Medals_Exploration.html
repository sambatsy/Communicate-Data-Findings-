def pie_chart(column):
    #quick function that visualizes a pie chart
    sorted_values = w_df[column].value_counts()

    w_df.groupby(column).size().plot(kind='pie',autopct=number, colors=pie_chart_colors)

    fig = plt.gcf()

def pointplot(col,n,title):
    ax = sb.pointplot(data = w_df, x = col, y = 'Age of Athlete', hue = 'Medal', order=w_df[col].value_counts().head(n).index,
                  dodge = 0.3, linestyles = "")
    sb.set(rc={'figure.figsize':(12.7,9.27)})
    plt.title(title)
